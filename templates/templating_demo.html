Jinja Templating Examples

for loop with Jinja

{% for o in output %}

{{o}}

{% endfor %}

if statement with Jinja

{% if output[3]|int > 10 %}

42 is greater than 10.

{% endif %}

replace( ) with Jinja

{{output[0]|replace('good', 'Owen Wilson')}}

loop through a string with .split( ) and make it a list

    {% for lang in output[1].split(',') %}
  • {{lang}}
    • {% endfor %}
